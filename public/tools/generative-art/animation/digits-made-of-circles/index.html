var group

function displayNumber(array) {
    if (group) {
        group.remove()
    }
    group = new Group()
    for (var x = 0; x < maxX; x++) {
        for (var y = 0; y < maxY; y++) {
            var circle = new Path.Circle({
                radius: 15,
                fillColor: 'red',
                center: [view.center.x + gap * x, view.center.y + gap * y],
                opacity: array[y][x],
                parent: group,
            })
            circle.opacity += 0.2
        }
    }
    return group
}
