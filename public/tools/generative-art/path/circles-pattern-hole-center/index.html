function createCircles(max, distanceFromCenter){
    var angle = 360/max
    var group = new Group()
    for (var i = 0; i < max; i++){
        var circle = new Path.Circle({
            radius: 10,
            fillColor: 'blue',
            center: [view.size.width/2, view.size.height/2 + distanceFromCenter],
            parent: group,
            applyMatrix: false
        })
    
 
        circle.rotate(angle * i, view.center)

    }

 

}
